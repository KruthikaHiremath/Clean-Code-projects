function checkDay(dayAbbreviation) { switch (dayAbbreviation) { case "Mon": console.log("Monday"); break; case "Tue": console.log("Tuesday"); break; case "Wed": console.log("Wednesday"); break; case "Thu": console.log("Thursday"); break; case "Fri": console.log("Friday"); break; case "Sat": console.log("Saturday"); break; case "Sun": console.log("Sunday"); break; default: console.log("Invalid day abbreviation"); } } checkDay("Mon"); // Output: Monday checkDay("Thu"); // Output: Thursday checkDay("Sat"); // Output: Saturday checkDay("ABC"); // Output: Invalid day abbreviation
