function checkGrade(score) { if (score >= 90) { console.log("A"); } else if (score >= 80 && score <= 89) { console.log("B"); } else if (score >= 70 && score <= 79) { console.log("C"); } else if (score >= 60 && score <= 69) { console.log("D"); } else { console.log("F"); } } checkGrade(95); // Output: A checkGrade(85); // Output: B checkGrade(75); // Output: C checkGrade(65); // Output: D checkGrade(55); // Output: F
